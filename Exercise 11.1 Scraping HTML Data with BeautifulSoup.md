ctx.verify_mode = ssl.CERT_NONE

url = input('Enter - ')
html = urllib.request.urlopen(url, context=ctx).read()
soup = BeautifulSoup(html, 'html.parser')

counts = dict()
tags = soup('span')
b = 0
a = 0
for tag in tags:
    #print(tag.contents[0])
    x = tag.contents[0]
    x = float(x)
    a = a + x
